88 def generate_keys(self):
89 """
90 Generate RSA key pair and store them in the specified locations.
91 """
92 pin = self.pin_input.text()
93 confirm_pin = self.confirm_pin_input.text()
94 private_key_path_folder = self.private_key_path.text()
95 public_key_path = self.public_key_path.text()
96
97 # Validate inputs
98 if not pin or len(pin) < 6 or not pin.isdigit():
99 QMessageBox.warning(self, "Invalid PIN", "PIN must be at least 6 digits.")
100 return
101
102 if pin != confirm_pin:
103 QMessageBox.warning(self, "PIN Mismatch", "PINs do not match.")
104 return
105
106 if not private_key_path_folder:
107 QMessageBox.warning(self, "No Private Key Location", "Please select a location for the private key.")
108 return
109
110 if not public_key_path:
111 QMessageBox.warning(self, "No Public Key Location", "Please select a location for the public key.")
112 return
113
114 try:
115 # Generate RSA key pair
116 self.status_label.setText("Generating RSA key pair...")
117 private_key, public_key = generate_rsa_keypair()
118
119 # Encrypt private key with PIN
120 self.status_label.setText("Encrypting private key...")
121 encrypted_private_key = encrypt_private_key(private_key, pin)
122
123 # Save private key to selected location
124 private_key_file_path = os.path.join(private_key_path_folder, "private_key.enc")
125 with open(private_key_file_path, 'wb') as f:
126 f.write(encrypted_private_key)
127
128 # Save public key to local storage
129 save_public_key(public_key, public_key_path)
130
131 self.status_label.setText("Keys generated and stored successfully!")
132 QMessageBox.information(self, "Success",
133 "RSA key pair generated successfully.\n"
134 f"Private key stored at: {private_key_file_path}\n"
135 f"Public key stored at: {os.path.join(public_key_path, 'public_key.pem')}")
136
137 except Exception as e:
138 self.status_label.setText(f"Error: {str(e)}")
139 QMessageBox.critical(self, "Error", f"Failed to generate keys: {str(e)}")
