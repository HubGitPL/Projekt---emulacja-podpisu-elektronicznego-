12def sign_pdf(input_path, output_path, private_key_pem):
13 """
14 Sign a PDF document and embed the signature.
15
16 Args:
17 input_path (str): Path to the input PDF
18 output_path (str): Path to save the signed PDF
19 private_key_pem (bytes): PEM-encoded private key
20 """
21 print("\n=== DETAILED PDF SIGNING DEBUGGING ===")
22 print(f"Input PDF: {input_path}")
23 print(f"Output PDF: {output_path}")
24
25 try:
26 # Read the PDF and calculate initial hash
27 with open(input_path, 'rb') as f:
28 pdf_content = f.read()
29 initial_hash = hashlib.sha256(pdf_content).digest()
30
31 # Read the PDF for processing
32 with open(input_path, 'rb') as f:
33 try:
34 pdf_reader = PdfReader(f)
35 except Exception as e:
36 print(f"Error reading PDF: {str(e)}")
37 raise ValueError("The PDF file appears to be corrupted or invalid. Please try with a different PDF file.")
38
39 pdf_writer = PdfWriter()
40
41 print(f"Original PDF has {len(pdf_reader.pages)} pages")
42
43 # Copy all pages from the original PDF
44 for page_num in range(len(pdf_reader.pages)):
45 try:
46 page = pdf_reader.pages[page_num]
47 pdf_writer.add_page(page)
48 except Exception as e:
49 print(f"Error processing page {page_num + 1}: {str(e)}")
50 raise ValueError(f"Error processing page {page_num + 1}. The PDF may be corrupted.")
51
52 # Calculate hash of the PDF content
53 pdf_hash = hashlib.sha256()
54 print("Calculating document hash for signing...")
55
56 # Create a buffer with the PDF content
57 pdf_buffer = io.BytesIO()
58 pdf_writer.write(pdf_buffer)
59 pdf_content_to_hash = pdf_buffer.getvalue()
60
61 # Hash the PDF content
62 pdf_hash.update(pdf_content_to_hash)
63 print(f"PDF content length: {len(pdf_content_to_hash)} bytes")
64
65 # Add the initial hash to the content hash
66 pdf_hash.update(initial_hash)
67
68 print(f"DEBUG: Signing - PDF hash: {pdf_hash.hexdigest()}")
69 print(f"Hash bytes length: {len(pdf_hash.digest())} bytes")
70
71 # Sign the hash
72 print("\nGenerating signature...")
73 signature = sign_data(pdf_hash.digest(), private_key_pem)
74 print(f"Signature bytes length: {len(signature)} bytes")
75 print(f"Signature hex: {signature.hex()[:64]}...")
76
77 # Add metadata
78 print("\nAdding document metadata...")
79 metadata = {
80 '/PAdES-Signature': 'True',
81 '/SignatureDate': f"{os.path.basename(input_path)}",
82 '/SignatureType': 'RSA-SHA256',
83 '/Signature': signature.hex(), # Store signature in metadata
84 '/InitialHash': initial_hash.hex(), # Store initial hash in metadata
85 '/Title': pdf_reader.metadata.get('/Title', '') + ' (Digitally Signed)',
86 '/Author': pdf_reader.metadata.get('/Author', ''),
87 '/Subject': pdf_reader.metadata.get('/Subject', ''),
88 '/Keywords': pdf_reader.metadata.get('/Keywords', ''),
89 '/Creator': pdf_reader.metadata.get('/Creator', ''),
90 '/Producer': pdf_reader.metadata.get('/Producer', ''),
91 '/CreationDate': pdf_reader.metadata.get('/CreationDate', ''),
92 '/ModDate': pdf_reader.metadata.get('/ModDate', '')
93 }
94 pdf_writer.add_metadata(metadata)
95 print(f"Added metadata keys: {list(metadata.keys())}")
96
97 # Save the signed PDF
98 print(f"\nSaving signed PDF to {output_path}...")
99 try:
100 with open(output_path, 'wb') as output_file:
101 pdf_writer.write(output_file)
102 print("PDF signed successfully")
103 except Exception as e:
104 print(f"Error saving signed PDF: {str(e)}")
105 raise ValueError("Error saving the signed PDF. Please check if you have write permissions to the output location.")
106 except Exception as e:
107 print(f"Error during PDF signing process: {str(e)}")
108 raise ValueError(f"Failed to sign document: {str(e)}")

110def create_signature_page(signature, initial_hash):
111 """
112 Create a PDF page containing the signature information.
113
114 Args:
115 signature (bytes): The digital signature
116 initial_hash (bytes): Hash of the original PDF content
117
118 Returns:
119 io.BytesIO: PDF page as a bytes buffer
120 """
121 print("Creating signature page with signature details...")
122 buffer = io.BytesIO()
123 c = canvas.Canvas(buffer, pagesize=letter)
124 width, height = letter
125
126 # Add title
127 c.setFont("Helvetica-Bold", 16)
128 c.drawString(72, height - 72, "Digital Signature")
129
130 # Add signature information
131 c.setFont("Helvetica", 12)
132 c.drawString(72, height - 100, "This document has been digitally signed.")
133 c.drawString(72, height - 120, "Signature Algorithm: RSA-SHA256")
134 c.drawString(72, height - 140, "Document Hash: " + initial_hash.hex()[:16])
135 c.drawString(72, height - 160, "Signature Hash: " + hashlib.sha256(signature).hexdigest()[:16])
136
137 # Add signature value visualization (first 64 chars of hex representation)
138 c.setFont("Courier", 10)
139 c.drawString(72, height - 200, "Signature Value (Visualization):")
140
141 # Split signature into multiple lines
142 line_length = 64
143 print("Adding signature visualization to page text:")
144 for i in range(0, min(len(signature.hex()), 256), line_length):
145 line_text = signature.hex()[i:i+line_length]
146 y_pos = height - 220 - (i // line_length * 15)
147 c.drawString(72, y_pos, line_text)
148 print(f" Line at y={y_pos}: {line_text[:16]}...")
149
150 c.save()
151 buffer.seek(0)
152 return buffer

154def verify_pdf_signature(pdf_path, public_key_pem):
155 """
156 Verify the signature of a signed PDF document.
157
158 Args:
159 pdf_path (str): Path to the signed PDF
160 public_key_pem (bytes): PEM-encoded public key
161
162 Returns:
163 bool: True if signature is valid, False otherwise
164 """
165 with open(pdf_path, 'rb') as f:
166 pdf_content = f.read()
167 pdf_reader = PdfReader(io.BytesIO(pdf_content))
168
169 print("\n=== DETAILED SIGNATURE VERIFICATION DEBUGGING ===")
170 print(f"PDF File: {pdf_path}")
171
172 # Check if this is a signed PDF
173 if '/PAdES-Signature' not in pdf_reader.metadata:
174 print("ERROR: No PAdES-Signature metadata found!")
175 raise ValueError("This PDF does not contain a PAdES signature")
176 else:
177 print(f"PAdES-Signature metadata found: {pdf_reader.metadata['/PAdES-Signature']}")
178
179 # Extract the signature from the metadata
180 if '/Signature' not in pdf_reader.metadata:
181 print("ERROR: No signature found in metadata!")
182 raise ValueError("No signature found in document metadata")
183
184 signature_hex = pdf_reader.metadata['/Signature']
185 print(f"Found signature in metadata: {signature_hex[:16]}...")
186
187 # Get initial hash from metadata
188 if '/InitialHash' not in pdf_reader.metadata:
189 print("ERROR: No initial hash found in metadata!")
190 raise ValueError("No initial hash found in document metadata")
191
192 initial_hash_hex = pdf_reader.metadata['/InitialHash']
193 print(f"Found initial hash in metadata: {initial_hash_hex[:16]}...")
194 initial_hash = bytes.fromhex(initial_hash_hex)
195
196 try:
197 # Convert hex to bytes
198 print("Converting signature from hex to bytes...")
199 signature = bytes.fromhex(signature_hex)
200 print(f"Signature bytes length: {len(signature)} bytes")
201 print(f"Signature first 16 bytes: {signature.hex()[:32]}")
202
203 # Now calculate the hash of the content
204 print("\nCalculating document hash...")
205 pdf_hash = hashlib.sha256()
206
207 # Create a new PDF writer to get the content without metadata
208 pdf_writer = PdfWriter()
209
210 # Copy all pages from the original PDF
211 for page in pdf_reader.pages:
212 pdf_writer.add_page(page)
213
214 # Get the PDF content without metadata
215 pdf_buffer = io.BytesIO()
216 pdf_writer.write(pdf_buffer)
217 pdf_content_to_hash = pdf_buffer.getvalue()
218
219 # Hash the PDF content
220 pdf_hash.update(pdf_content_to_hash)
221 print(f"PDF content length: {len(pdf_content_to_hash)} bytes")
222
223 # Add the initial hash to the content hash
224 pdf_hash.update(initial_hash)
225 print("Added initial hash to content hash")
226
227 print(f"Final hash digest: {pdf_hash.hexdigest()}")
228 print(f"Hash bytes length: {len(pdf_hash.digest())} bytes")
229
230 # Verify the signature
231 print("\nAttempting signature verification...")
232 print(f"Data hash to verify: {pdf_hash.hexdigest()}")
233 print(f"Data hash bytes length: {len(pdf_hash.digest())} bytes")
234
235 result = verify_signature(pdf_hash.digest(), signature, public_key_pem)
236 print(f"DEBUG: Verification result: {result}")
237
238 return result
239
240 except Exception as e:
241 print(f"DEBUG: Verification exception: {str(e)}")
242 raise ValueError(f"Failed to verify signature: {str(e)}")
