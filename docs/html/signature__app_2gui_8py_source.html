183 def sign_document(self):
184 """
185 Sign the selected PDF document.
186 """
187 pdf_path = self.pdf_path.text()
188 output_path = self.output_path.text()
189 pin = self.pin_input.text()
190
191 # Validate inputs
192 if not pdf_path:
193 QMessageBox.warning(self, "No PDF Selected", "Please select a PDF document to sign.")
194 return
195
196 if not output_path:
197 QMessageBox.warning(self, "No Output Location", "Please select an output location for the signed PDF.")
198 return
199
200 if not pin:
201 QMessageBox.warning(self, "No PIN Entered", "Please enter your PIN.")
202 return
203
204 # Check if private key is on USB and USB is still connected
205 if not self.current_usb_path or not self.encrypted_key_data:
206 QMessageBox.critical(self, "USB Disconnected",
207 "The USB drive containing your private key has been disconnected. "
208 "Please reconnect it to sign the document.")
209 return
210
211 try:
212 # Show progress
213 self.progress_bar.setVisible(True)
214 self.progress_bar.setValue(10)
215
216 # Decrypt the private key
217 private_key = decrypt_private_key(self.encrypted_key_data, pin)
218 self.key_status.setText("Key Status: Loaded")
219 self.key_status.setStyleSheet("color: green;")
220
221 self.progress_bar.setValue(30)
222
223 # Sign the PDF
224 sign_pdf(pdf_path, output_path, private_key)
225
226 self.progress_bar.setValue(100)
227
228 # Hide progress bar after a delay
229 QTimer.singleShot(1000, lambda: self.progress_bar.setVisible(False))
230
231 QMessageBox.information(self, "Success", "Document signed successfully!")
232
233 except Exception as e:
234 self.progress_bar.setVisible(False)
235 error_msg = str(e)
236 if "Invalid PIN" in error_msg:
237 QMessageBox.critical(self, "Invalid PIN",
238 "The PIN you entered is incorrect. Please try again.")
239 else:
240 QMessageBox.critical(self, "Error", f"Failed to sign document: {error_msg}")

326 def verify_signature(self):
327 """
328 Verify the signature of the selected PDF document.
329 """
330 pdf_path = self.signed_pdf_path.text()
331 public_key_path = self.public_key_path.text()
332
333 # Validate inputs
334 if not pdf_path:
335 QMessageBox.warning(self, "No PDF Selected", "Please select a signed PDF document to verify.")
336 return
337
338 if not public_key_path:
339 QMessageBox.warning(self, "No Public Key", "Please select the public key file.")
340 return
341
342 try:
343 # Load public key
344 with open(public_key_path, 'rb') as f:
345 public_key_data = f.read()
346
347 # Verify the signature
348 try:
349 is_valid = verify_pdf_signature(pdf_path, public_key_data)
350
351 if is_valid:
352 self.verification_result.setText("✅ Signature is valid!")
353 self.verification_result.setStyleSheet("color: green; font-weight: bold;")
354 else:
355 self.verification_result.setText("❌ Signature is invalid!")
356 self.verification_result.setStyleSheet("color: red; font-weight: bold;")
357 QMessageBox.warning(self, "Invalid Signature",
358 "The signature could not be verified. This might be because:\n\n"
359 "1. The document was modified after signing\n"
360 "2. The public key doesn't match the private key used to sign\n"
361 "3. The signature was corrupted or improperly embedded")
362 except ValueError as ve:
