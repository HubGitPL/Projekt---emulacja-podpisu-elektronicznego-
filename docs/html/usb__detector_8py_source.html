27 def run(self):
28 """
29 Main thread loop for monitoring USB drives.
30 """
31 # Check for existing drives with private keys when monitoring starts
32 for drive in self.connected_drives:
33 self.check_drive_for_key(drive)
34

36 current_drives = self.get_removable_drives()

43

53 def check_drive_for_key(self, drive_path):
54 """
55 Check if a drive contains a private key file and load it if found.
56
57 Args:
58 drive_path (str): Path to the drive to check
59 """
60 key_path = os.path.join(drive_path, "private_key.enc")
61 if os.path.exists(key_path):
62 try:
63 with open(key_path, 'rb') as f:
64 encrypted_key_data = f.read()
65 self.status_update.emit(f"Private key found on {drive_path}")
66 self.usb_connected.emit(drive_path, encrypted_key_data)
67 except Exception as e:
68 self.status_update.emit(f"Error reading private key: {str(e)}")

70 def get_removable_drives(self):
71 """
72 Get a list of removable drives.
73
74 Returns:
75 list: List of drive paths
76 """
77 drives = []
78
79 if self.system == "Darwin": # macOS
80 try:
81 if os.path.exists("/Volumes"):
82 volumes = os.listdir("/Volumes")
83 for volume in volumes:
84 volume_path = os.path.join("/Volumes", volume)
85
86 # Skip if not a mount point
87 if not os.path.ismount(volume_path):
88 continue
89
90 # Skip system volumes and Time Machine backups
91 if any(x in volume for x in ["Macintosh", "System", "Time Machine", "com.apple"]):
92 continue
93
94 # Get filesystem info
95 try:
96 st = os.statvfs(volume_path)
97 # Check if volume is writable
98 if (st.f_flag & os.ST_RDONLY) == 0:
99 drives.append(volume_path)
100 except:
101 pass
102 except Exception as e:
103 self.status_update.emit(f"Error detecting drives: {str(e)}")
104 else:
105 # For other systems (Windows/Linux), use psutil
106 for partition in psutil.disk_partitions():
107 if 'removable' in partition.opts or 'cdrom' in partition.opts:
108 drives.append(partition.mountpoint)
109
110 return drives
